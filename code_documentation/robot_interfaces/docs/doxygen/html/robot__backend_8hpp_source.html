242  void loop()
243  {
244  const double start_time =
245  real_time_tools::Timer::get_current_time_sec();
246 
247  // wait until first desired_action was received
248  // ----------------------------
249  while (!has_shutdown_request() &&
250  !robot_data_->desired_action->wait_for_timeindex(0, 0.1))
251  {
252  const double now = real_time_tools::Timer::get_current_time_sec();
253  if (now - start_time > first_action_timeout_)
254  {
255  Status status;
256  status.set_error(Status::ErrorStatus::BACKEND_ERROR,
257  "First action was not provided in time");
258 
259  robot_data_->status->append(status);
260 
261  std::cerr << "Error: " << status.error_message
262  << "\nRobot is shut down." << std::endl;
263 
264  request_shutdown();
265  break;
266  }
267  }
268 
269  for (long int t = 0; !has_shutdown_request(); t++)
270  {
271  // TODO: figure out latency stuff!!
272 
273  Status status;
274 
275  if (max_number_of_actions_ > 0 && t >= max_number_of_actions_)
276  {
277  // TODO this is not really an error
278  status.set_error(Status::ErrorStatus::BACKEND_ERROR,
279  "Maximum number of actions reached.");
280  }
281 
282  timer_.start();
283 
284  // get latest observation from robot and append it to robot_data_
285  Observation observation = robot_driver_->get_latest_observation();
286  timer_.checkpoint("get observation");
287 
288  robot_data_->observation->append(observation);
289  // TODO: for some reason this sometimes takes more than 2 ms
290  // i think this may be due to a non-realtime thread blocking the
291  // timeseries. this is in fact an issue, we might have to
292  // duplicate all the timeseries and have a realtime thread
293  // writing back and forth
294  timer_.checkpoint("append observation");
295 
296  // If real time mode is enabled the next action needs to be provided
297  // in time. If this is not the case, optionally repeat the previous
298  // action or raise an error.
299  if (real_time_mode_ &&
300  robot_data_->desired_action->newest_timeindex() < t)
301  {
302  uint32_t action_repetitions =
303  robot_data_->status->newest_element().action_repetitions;
304 
305  if (action_repetitions < max_action_repetitions_)
306  {
307  robot_data_->desired_action->append(
308  robot_data_->desired_action->newest_element());
309  status.action_repetitions = action_repetitions + 1;
310  }
311  else
312  {
313  // No action provided and number of allowed repetitions
314  // of the previous action is exceeded --> Error
315  status.set_error(Status::ErrorStatus::BACKEND_ERROR,
316  "Next action was not provided in time");
317  }
318  }
319 
320  std::string driver_error_msg = robot_driver_->get_error();
321  if (!driver_error_msg.empty())
322  {
323  status.set_error(Status::ErrorStatus::DRIVER_ERROR,
324  driver_error_msg);
325  }
326 
327  robot_data_->status->append(status);
328 
329  // if there is an error, shut robot down and stop loop
330  if (status.error_status != Status::ErrorStatus::NO_ERROR)
331  {
332  std::cerr << "Error: " << status.error_message
333  << "\nRobot is shut down." << std::endl;
334  break;
335  }
336  timer_.checkpoint("status");
337 
338  // early exit if destructor has been called
339  while (!has_shutdown_request() &&
340  !robot_data_->desired_action->wait_for_timeindex(t, 0.1))
341  {
342  }
343  if (has_shutdown_request())
344  {
345  break;
346  }
347 
348  Action desired_action = (*robot_data_->desired_action)[t];
349  timer_.checkpoint("get action");
350 
351  Action applied_action = robot_driver_->apply_action(desired_action);
352  timer_.checkpoint("apply action");
353 
354  robot_data_->applied_action->append(applied_action);
355  timer_.checkpoint("append applied action");
356 
357  if (t % 5000 == 0 && t > 0)
358  {
359  timer_.print_statistics();
360  }
361  }
362 
363  robot_driver_->shutdown();
364  loop_is_running_ = false;
365  }
