134 private:

142  void loop()
143  {
144  auto t = sensor_data_->observation->newest_timeindex();
145 
146  while (enabled_)
147  {
148  try
149  {
150  auto timestamp = sensor_data_->observation->timestamp_ms(t);
151  auto observation = (*sensor_data_->observation)[t];
152  buffer_.push_back(std::make_tuple(timestamp, observation));
153  }
154  catch (const std::invalid_argument &e)
155  {
156  std::cerr << "ERROR: " << e.what() << "\nSkip observation."
157  << std::endl;
158  }
159  t++;
160 
161  // Stop logging if buffer limit is reached
162  if (buffer_.size() >= buffer_limit_)
163  {
164  std::cerr << "WARNING: SensorLogger buffer limit is reached. "
165  "Stop logging."
166  << std::endl;
167  enabled_ = false;
168  }
169  }
170  }
