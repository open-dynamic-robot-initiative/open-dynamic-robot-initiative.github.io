109  void stop()
110  {
111  // This is a bit complicated: In any case, join the thread if it is
112  // joinable. However, only write the remaining data to file if the
113  // logging thread is actually running at the moment stop() is called.
114  bool still_running = is_running_;
115 
116  stop_was_called_ = true;
117  if (thread_.joinable())
118  {
119  thread_.join();
120  }
121 
122  if (still_running)
123  {
124  append_robot_data_to_file(index_, block_size_);
125  }

223  void append_names_to_header(
224  const std::string &identifier,
225  const std::vector<std::string> &field_name,
226  const std::vector<std::vector<double>> &field_data,
227  std::vector<std::string> &header)
228  {
229  for (size_t i = 0; i < field_name.size(); i++)
230  {
231  if (field_data[i].size() == 1)
232  {
233  std::string temp = identifier + "_" + field_name[i];
234  header.push_back(temp);
235  }
236  else
237  {
238  for (size_t j = 0; j < field_data[i].size(); j++)
239  {
240  std::string temp = identifier + "_" + field_name[i] + "_" +
241  std::to_string(j);
242  header.push_back(temp);
243  }
244  }
245  }

270  void append_robot_data_to_file(long int start_index, long int block_size)
271  {
272  output_file_.open(output_file_name_, std::ios_base::app);
273  output_file_.precision(27);
274 
275  for (long int t = start_index;
276  t < std::min(start_index + block_size,
277  logger_data_->observation->newest_timeindex());
278  t++)
279  {
280  try
281  {
282  Action applied_action = (*logger_data_->applied_action)[t];
283  Action desired_action = (*logger_data_->desired_action)[t];
284  Observation observation = (*logger_data_->observation)[t];
285  Status status = (*logger_data_->status)[t];
286  auto timestamp = logger_data_->observation->timestamp_s(t);
287 
288  output_file_ << t << " " << timestamp << " ";
289  append_field_data_to_file(status.get_data());
290  append_field_data_to_file(observation.get_data());
291  append_field_data_to_file(applied_action.get_data());
292  append_field_data_to_file(desired_action.get_data());
293  output_file_ << std::endl;
294  }
295 
296  catch (const std::invalid_argument &e)
297  {
298  auto t_oldest = logger_data_->observation->oldest_timeindex();
299  auto diff = t_oldest - t;
300 
301  std::cout << "Warning: Trying to log time step " << t
302  << " which is not in the buffer anymore. Skip "
303  << diff << " steps to time step " << (t_oldest + 1)
304  << std::endl;
305 
306  // Note that t is incremented in the next iteration, so logging
307  // continues from t_oldest + 1. This means that one potentially
308  // available step is dropped but it lowers the risk that the
309  // same issue happens immediately again as t_oldest may drop out
310  // of the buffer until it is processed in the next iteration.
311  t = t_oldest;
312  }
313  }
314 
315  output_file_.close();
316  }

341  void loop()
342  {
343  is_running_ = true;
344 
345  write_header_to_file();
346 
347  while (!stop_was_called_ &&
348  !(logger_data_->desired_action->length() > 0))
349  {
350  std::this_thread::sleep_for(std::chrono::milliseconds(100));
351  }
352 
353  index_ = logger_data_->observation->newest_timeindex();
354 
355  while (!stop_was_called_)
356  {
357  if (index_ + block_size_ <=
358  logger_data_->observation->newest_timeindex())
359  {
360 #ifdef VERBOSE
361  auto t1 = std::chrono::high_resolution_clock::now();
362 #endif
363 
364  append_robot_data_to_file(index_, block_size_);
365 
366  index_ += block_size_;
367 
368 #ifdef VERBOSE
369  auto t2 = std::chrono::high_resolution_clock::now();
370  auto duration =
371  std::chrono::duration_cast<std::chrono::microseconds>(t2 -
372  t1)
373  .count();
374 
375  // to print the time taken for one block of data to be logged.
376  std::cout << "Time taken for one block of data to be logged: "
377  << duration << std::endl;
378 #endif
379  }
380  }
381 
382  is_running_ = false;
383  }
